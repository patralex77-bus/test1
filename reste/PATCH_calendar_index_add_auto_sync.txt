# === OPTIONAL PATCH for app/blueprints/calendar/templates/calendar/index.html ===
# Add this <script> near the bottom of the calendar page (before </body>)
# It pulls orders from /orders/api/list and stores them where the calendar already reads.
<script>
(async function fillOrdersFromServer(){
  try{
    const res = await fetch('/orders/api/list');
    const data = await res.json();
    const orders = Array.isArray(data.orders) ? data.orders : [];
    try{
      localStorage.setItem('busops:orders_v1', JSON.stringify(orders));
      localStorage.setItem('orders_fallback_store_v1', JSON.stringify(orders));
    }catch(e){}
    try{
      if(window.OrdersShared && typeof window.OrdersShared.ingest === 'function'){
        window.OrdersShared.ingest(orders);
      }
    }catch(e){}
    try{
      if(window.Calendar && typeof window.Calendar.rerender==='function'){ window.Calendar.rerender(); }
    }catch(e){}
  }catch(e){}
})();
</script>
# === END OPTIONAL PATCH ===
